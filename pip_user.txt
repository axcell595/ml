# 1) See which global packages were installed via pip
pip3 list --format=columns | awk '$NF=="pip"{print $1}'

# 2) Remove all global pip installs (dry-run: print first, then run uninstall)
sudo pip3 uninstall -y $(pip3 list --format=columns | awk '$NF=="pip"{print $1}')
